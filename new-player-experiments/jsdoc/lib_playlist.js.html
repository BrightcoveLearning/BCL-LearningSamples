    const playbackApi = require('./playbackApi');
const video = require('./video');
const url = require('./url');
const constants = require('../constants');
const vars = require('./vars');

function noVideos () {
  return {
    videos: [],
  };
}

/**
 * Retrieves a playlist from the Playback or YouTube API
 */
async function getPlaylist (type, id, experience, offset, limit) {
  const { accountId, playbackApiBaseUrl, youtubeApiBaseUrl, policyKey, youtubeKey } = experience;
  if (type === constants.video.sources.youtube) {
    const page = offset || '';
    const params = url.toQueryString({ page, count: limit });
    const youtubeBaseUrl = playbackApi.getYouTubeBaseUrl(youtubeApiBaseUrl, accountId);

    const { items, nextPageToken } = await playbackApi.get(`${youtubeBaseUrl}/playlists/${id}${params}`, youtubeKey);
    const videos = items.map(item => Object.assign(item, {
      id: `${type}:${item.id}`,
    }));
    return { videos, nextPageToken };
  }
  if (type === constants.video.sources.ref) {
    id = `ref:${id}`;
  }
  const baseUrl = playbackApi.getPlaybackApiBaseUrl(playbackApiBaseUrl, accountId);
  const params = url.toQueryString({ offset, limit });
  return await playbackApi.get(`${baseUrl}/playlists/${id}${params}`, policyKey);
}

async function fetch (playlistId, experience, offset, limit) {
  const { type, id } = video.getTypeAndId(playlistId);
  //Interpolate variables in the ID, fall back to the defaults if the ID is falsy
  const defaultsId = vars.defaults(id);
  const interpolatedId = vars.interpolate(id) || defaultsId;

  if (!interpolatedId) {
    return noVideos();
  }

  try {
    return await getPlaylist(type, interpolatedId, experience, offset, limit);
  } catch (err) {
    //If the interpolatedId and defaultsId are the same, we shouldn't call the playback API again,
    //just rethrow the error.
    if (interpolatedId === defaultsId) {
      throw err;
    }
    if (!defaultsId) {
      return noVideos();
    }
    return await getPlaylist(type, defaultsId, experience, offset, limit);
  }
}

module.exports = { fetch };
