class Tracker {
  constructor (experience, activeViewName) {
    this.experience = experience;
    this.activeViewName = activeViewName;

    this.events = new Events();
    this.lookupMap = {};
  }

  trackEvent (/* action, metadata */) {
    throw new Error('Not implemented');
  }

  /**
   * Wraps event handlers and tracks the specified action when the event handler is triggered
   * metadata could contain any or all of the following fields:
   *   videoId, lookup, label
   */
  track (action, next, metadata) {
    return (...args) => {
      this.trackEvent(action, metadata);
      if (next) {
        return next(...args);
      }
    };
  }

  /**
   * Helper for tracking actions inside a Promise.then() function
   */
  then (action, metadata) {
    return data => {
      this.trackEvent(action, metadata);
      return data;
    };
  }

  /**
   * Helper for tracking actions inside a Promise.catch() function
   */
  _catch (action, metadata) {
    return err => {
      this.trackEvent(action, metadata);
      throw err;
    };
  }

  convertActionText (action) {
    const parts = action.split('.');

    let lookup = this.lookupMap;
    let i;
    const partsLength = parts.length;
    for (i = 0; i < partsLength; i++) {
      const part = parts[i];
      if (lookup[part] == null) {
        break;
      }
      if (lookup[part] === false) {
        return false;
      }
      lookup = lookup[part];
    }
    if (typeof lookup === 'function') {
      return lookup(...parts.slice(i));
    }
    if (partsLength === i) {
      return lookup;
    }
    return action;
  }
}

module.exports = Tracker;
