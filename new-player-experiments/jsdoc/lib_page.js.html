    const window = require('global/window');
const document = require('global/document');

function wait (milliseconds) {
  return new Promise((resolve) => setTimeout(resolve, milliseconds));
}

/**
 * Gets the URL of the page we're embedded on
 */
function getLocation () {
  try {
    //Try to get the top window location
    return window.top.location.href;
  } catch (e) {
    //IFramed, try to find the top-most document.referrer
    let { referrer } = document;
    let { parent } = window;
    while (parent) {
      try {
        referrer = parent.document.referrer || referrer;
        parent = parent.parent;
      } catch (e) {
        //No higher access
        break;
      }
    }
    return referrer;
  }
}

/**
 * Gets the referrer of the page we're embedded on
 */
function getReferrer () {
  try {
    //Try to get the top window referrer
    return window.top.document.referrer;
  } catch (e) {
    //IFramed, no access to the referrer
    return null;
  }
}

function isAccessible () {
  try {
    return !!window.top.document;
  } catch (e) {
    return false;
  }
}

function getQuery () {
  const href = window.location.href;
  if (!href) {
    return {};
  }
  if (href.split('?').length < 2) {
    return {};
  }
  const queryString = href.split('?').pop();
  const tuples = queryString.split('&');

  return tuples.reduce((query, tuple) => {
    const [key, value] = tuple.split('=');
    query[decodeURIComponent(key)] = value ? decodeURIComponent(value) : null;
    return query;
  }, {});
}

function getEmbedDimensions () {
  return {
    height: window.innerHeight,
    width: window.innerWidth,
  };
}

function transferGlobals (parent, target, keys = []) {
  return wait(1)
    .then(() => {
      keys.forEach((key) => {
        if (!target[key]) {
          target[key] = parent[key];
        }
      });
    });
}

module.exports = {
  getLocation,
  getReferrer,
  isAccessible,
  getQuery,
  getEmbedDimensions,
  transferGlobals,
};
